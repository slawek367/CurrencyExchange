Wallet table:
CREATE TABLE Wallets (
	id INT(11) AUTO_INCREMENT PRIMARY KEY,
	user_id INT(11),
	BTC DECIMAL(20, 8),
	ETH DECIMAL(20, 8),
	LTC DECIMAL(20, 8),
	USD DECIMAL(20, 8),
	FOREIGN KEY (user_id) REFERENCES users(id)
);
